# 4. 다형성 높이기

## 1.4. 다형성 높이기

### 1.4.1. _each 함수 다형성 높이기
기존 `_each` 함수의 리스트 인자에 `null`을 넣으면 내부에서 `length`를 참조하고자 할때 에러가 난다.
```javascript
_each(null, console.log) // error
```
앞서 만든 함수 중에 `_get` 함수가 null 값이면 `undefined`를 반환하므로 어떻게 보면 `null` 체크를 하고 있다. `_each` 함수 내부에 `list.length` 대신 `undefined`가 들어가면 에러가 나지 않으므로 아래와 같이 수정하면 에러를 방지할수 있다.
```javascript
let _length = _get('length')
let _each = (list, iteratee) => {
	for(let i = 0, len = _length(list); i < len; i++) {
		iteratee(list[i]);
	}
}
```
위와 같이 수정하면 `_each` 함수로 만든 `_map`, `_filter` 함수 역시 에러를 피해갈 수 있다.
```javascript
console.log(
  _filter(null, v => v), // []
  _map(null, v => v), // []
)
```
함수형 프로그래밍에서는 에러가 났을때 흘려 보낼수 있는 전략을 취한다. 그런 이유로 함수들을 연속 실행 할 때, 언제 어디서 잘못된 값이 들어와도 특별히 에러가 나지 않고 흘려 보낼수 있으며, `underscore`, `lodash` 같은 라이브러리에서도 이런 전략을 채택하고 있다.
이런 식으로 로직이 흘러가게 되면 데이터 형을 체크하거나 특정 작업을 `trycatch` 구문으로 묶어주는 작업이 줄어든다.

### 1.4.2. _keys
자바스크립트 Object 객체의 내장 함수(메서드) 중 keys는 객체를 받아 해당 객체의 키 값만 리턴해주는 함수인데 null 값이 들어오게 되면 에러가 발생한다.
```javascript
console.log(Object.keys({ name: 'ID', age: 33 }));
console.log(Object.keys([1, 2, 3, 4]));
console.log(Object.keys(10));
// console.log(Object.keys(null)); // 에러
```
이때 _keys 함수를 만들어 문제를 해결할 수 있다.
```javascript
function _is_object(obj) {
  return typeof obj == 'object' && !!obj;
}
function _keys(obj) {
  return _is_object(obj) ? Object.keys(obj) : [];
}
```
```javascript
console.log(_keys({ name: 'ID', age: 33 }));
console.log(_keys([1, 2, 3, 4]));
console.log(_keys(10));
console.log(_keys(null));
```
이런 식으로 함수형 프로그래밍에서는 어떤 형태의 값이 들어오던지 그럴싸한 값을 리턴하는 식으로 다형성을 높여, 연속적인 함수 실행에 무리가 없도록 한다.
위 코드를 이용하면 _each 함수의 다형성을 높일 수 있다.
```javascript
_each({
  13: 'AB',
  19: 'CD',
  29: 'YD',
}, name => console.log(name))
// length 가 없으므로 아무런 일도 안일어난다.
```
```javascript
let _each = (list, iteratee) => {
	let keys = _keys(list) // keys 자체는 null 값이 와도 빈 배열을 뱉도록 준비돼 있다. 무조건 올바른 배열이 리턴되므로 length를 사용할 수 있다.
	for(let i = 0, len = keys.length; i < len; i++) {
		iteratee(list[keys[i]]);
	}
}
```
